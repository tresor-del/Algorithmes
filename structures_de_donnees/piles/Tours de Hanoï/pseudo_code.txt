    // Cas de base : Un seul disque est déplacé directement

    DEPLACER_DISQUES(n - 1, SOURCE, AUXILIAIRE, DESTINATION)

    AFFICHER "Déplacer le disque " + n + " de " + SOURCE + " vers " + DESTINATION

    // (en utilisant SOURCE comme pile temporaire)
